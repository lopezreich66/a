  <script>
    const params = new URLSearchParams(window.location.search);
    const produto = params.get('produto');
    const valor = parseFloat(params.get('valor'));
    const frete = parseFloat(params.get('frete'));
    const email = params.get('email');
    const cep = params.get('cep');
    const numero = params.get('numero');
    const complemento = params.get('complemento');

    const total = valor + frete;

    document.getElementById('summary').innerHTML = `
      <p><strong>Produto:</strong> ${produto}</p>
      <p><strong>Valor do produto:</strong> R$${valor.toFixed(2)}</p>
      <p><strong>Frete estimado:</strong> R$${frete.toFixed(2)}</p>
      <p><strong>Total a pagar:</strong> R$${total.toFixed(2)}</p>
      <p><strong>Email:</strong> ${email}</p>
      <p><strong>CEP:</strong> ${cep}</p>
      <p><strong>Número:</strong> ${numero}</p>
      ${complemento ? `<p><strong>Complemento:</strong> ${complemento}</p>` : ''}
    `;

    const API_URL = 'https://api.pushinpay.com.br/api/pix/cashIn';
    const TOKEN = '61382|C0izZLKkR0LdwJQzSbKFqylrgNkMBePNwRYobLZledd4ee14';

    async function gerarPixFinal() {
      document.getElementById('confirmBtn').style.display = 'none';

      const pixArea = document.getElementById('pixArea');
      pixArea.style.display = 'block';

      const valueEl = document.getElementById('pixValue');
      const qrEl = document.getElementById('qrImage');
      const codeEl = document.getElementById('pixCode');
      const timerEl = document.getElementById('timeLeft');

      valueEl.textContent = `Valor: R$${total.toFixed(2)} - ${produto}`;
      qrEl.src = '';
      codeEl.value = 'Gerando PIX... Aguarde';

      try {
        const response = await fetch(API_URL, {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${TOKEN}`,
            'Accept': 'application/json',
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            value: Math.round(total * 100),
            webhook_url: ''
          })
        });

        const data = await response.json();

        if (response.ok && data.qr_code_base64 && data.qr_code) {
          qrEl.src = data.qr_code_base64.includes('data:') ? data.qr_code_base64 : `data:image/png;base64,${data.qr_code_base64}`;
          codeEl.value = data.qr_code;

          let time = 900;
          const interval = setInterval(() => {
            time--;
            const min = Math.floor(time / 60);
            const sec = time % 60;
            timerEl.textContent = `${min}:${sec < 10 ? '0' : ''}${sec}`;
            if (time <= 0) {
              clearInterval(interval);
              timerEl.textContent = 'Tempo esgotado';
            }
          }, 1000);
        } else {
          codeEl.value = `Erro ao gerar PIX: ${data.message || 'Verifique token ou IP'}`;
        }
      } catch (error) {
        codeEl.value = `Falha na conexão: ${error.message}`;
        console.error(error);
      }
    }

    function copyPixCode() {
      const code = document.getElementById('pixCode');
      code.select();
      document.execCommand('copy');
      alert('Código PIX copiado!');
    }
